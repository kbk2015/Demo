import com.github.tomakehurst.wiremock.extension.Parameters;
import com.github.tomakehurst.wiremock.extension.requestfilter.RequestFilterAction;
import com.github.tomakehurst.wiremock.extension.requestfilter.RequestWrapper;
import com.github.tomakehurst.wiremock.extension.requestfilter.StubRequestFilter;

public class RequestLoggingExtension extends StubRequestFilter {

    @Override
    public RequestFilterAction filter(RequestWrapper request, Parameters parameters) {
        // Log the incoming request details
        System.out.println("Received request: " + request.getUrl());
        System.out.println("Headers: " + request.getAllHeaderKeys());
        System.out.println("Body: " + request.getBodyAsString());

        // You can perform additional actions with the request here if needed

        // Return RequestFilterAction.continueWith(request) to continue processing the request
        return RequestFilterAction.continueWith(request);
    }
}
